# 获取个人档案

说明：根据 fid 获取个人档案

# 获取方式： HTTP GET

# 发送参数:

{
	"fid": 10001
}

# 返回数据格式: JSON

# 接口地址(模拟)：http://localhost:3000/personFile

# 调用例子: http://localhost:3000/personFile

# 返回示例:

//院长、部门主管
{
    "success": true,
    "code": 200,
    "msg": "查询成功",
    "data": {
        "fid": 10005,
        "name": "吴一凡",
        "sex": "男",
        "age": 51,
        "phone": "15916100000",
        "email": "helloworld@qq.com",
        "positionId": 1,
        "positionName": "院长"
    }
}

//教务员
{
    "success": true,
    "code": 200,
    "msg": "查询成功",
    "data": {
        "fid": 10003,
        "name": "李四",
        "sex": "男",
        "age": 35,

        "email": "10001@qq.com",
        "sectionId": 1,
        "sectionName": "人事处",

# 人员档案录入

说明：录入人员档案，自动生成人员档案ID

# 获取方式： HTTP POST

# 发送参数:

{
	"name": "苏少权",
  	"sex": "男",
  	"age": "22",
  	"phone": "15811000000",
  	"email": "sushaoquanssb@qq.com",
  	"positionId": 4,
    /**********以下参数需判断后进行录入***********/
    //教师
    "stationId": 3,
  	"levelId": 6,
    //教务员
    "sectionId": 2
}

# 返回数据格式: JSON

# 接口地址(模拟)：http://localhost:3000/m_register

# 调用例子: http://localhost:3000/m_register

# 返回示例:

{
    "success": true,
    "code": 200,
    "msg": "注册成功，职工号10086"
}

{
    "success": false,
    "code": 200,
    "msg": "注册失败，请重试"
}
