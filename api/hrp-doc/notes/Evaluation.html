# 绩效考核管理

# 获取绩点记录情况

说明：获取所有教师绩点记录情况

# 获取方式： HTTP GET

# 发送参数:

{
	"page": 1,
    "year": "2020"
}

# 返回数据格式: JSON

# 接口地址(模拟)：http://localhost:3000/gpa_record

# 调用例子: http://localhost:3000/gpa_record

# 返回示例:

{
    "success": true,
    "code": 200,
    "msg": "查询成功",

        "data": [

                "isUpdate": 0,
                "isEvaluate": 0,
                "stationId": 3,
                "levelId": 4
            },
            {
                "loadRecordId": 6,
                "teachLoad": 200,
                "fid": 10023,

# 绩效考核

说明：对教师绩效进行考核

# 获取方式： HTTP POST

# 发送参数:

{
	"fid": 10006,
	"teachLoad": 408,
	"publicGpa": 70,
	"scientGpa": 5,
	"stationId": 4,
	"levelId": 6
}

# 返回数据格式: JSON

# 接口地址(模拟)：http://localhost:3000/evaluate

# 调用例子: http://localhost:3000/evaluate

# 返回示例:

{
    "success": false,
    "code": 200,
    "msg": "请确保教务处已录入课时量"
}

{
    "success": false,
    "code": 200,
    "msg": "考核失败，请重试"
}

{
    "success": true,
    "code": 200,
    "msg": "考核成功"
}

# 绩效再评

说明：对绩效重新进行评价

# 获取方式： HTTP POST

# 发送参数:

{
    "evaluateRecordId": 1, 
    "grade": 80   //这里设定的分值是最后得分
}

# 返回数据格式: JSON

# 接口地址(模拟)：http://localhost:3000/reEvaluate

# 调用例子: http://localhost:3000/reEvaluate

# 返回示例:

{
    "success": true,
    "code": 200,
    "msg": "绩效再评成功"
}

{
    "success": false,
    "code": 200,
    "msg": "绩效再评失败，请重试"
}
